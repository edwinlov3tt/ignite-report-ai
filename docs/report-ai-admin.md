## 1. Platform Knowledge Manager

This should be **separate from tactics** because platforms cut across all tactics (Facebook Retargeting, Facebook Prospecting, etc. all share Facebook quirks).

### Database Schema

```sql
-- Platforms (top level)
CREATE TABLE platforms (
  id TEXT PRIMARY KEY,
  name TEXT NOT NULL,              -- "Facebook/Meta"
  code TEXT UNIQUE NOT NULL,       -- "facebook"
  category TEXT NOT NULL,          -- "social", "search", "programmatic"
  logo_url TEXT,
  is_active BOOLEAN DEFAULT true,
  created_at TIMESTAMP DEFAULT NOW(),
  updated_at TIMESTAMP DEFAULT NOW()
);

-- Platform Quirks (the gold - what AI needs to know)
CREATE TABLE platform_quirks (
  id TEXT PRIMARY KEY,
  platform_id TEXT REFERENCES platforms(id),
  quirk_type TEXT NOT NULL,        -- "attribution", "targeting", "bidding", "reporting", "creative"
  title TEXT NOT NULL,             -- "Learning Phase Requirements"
  description TEXT NOT NULL,       -- Full explanation
  impact TEXT,                     -- "High", "Medium", "Low"
  ai_instruction TEXT,             -- How AI should handle this
  source TEXT,                     -- "Platform docs", "Buyer experience", "DCM discovery"
  contributed_by TEXT,             -- Who added this
  verified_by TEXT,                -- DCM who verified
  is_active BOOLEAN DEFAULT true,
  created_at TIMESTAMP DEFAULT NOW(),
  updated_at TIMESTAMP DEFAULT NOW()
);

-- Platform KPIs (what metrics matter by objective)
CREATE TABLE platform_kpis (
  id TEXT PRIMARY KEY,
  platform_id TEXT REFERENCES platforms(id),
  objective TEXT NOT NULL,         -- "awareness", "traffic", "conversions"
  primary_kpis JSONB NOT NULL,     -- ["cpm", "reach", "frequency"]
  secondary_kpis JSONB,            -- ["video_views", "brand_lift"]
  notes TEXT,
  created_at TIMESTAMP DEFAULT NOW()
);

-- Platform Warning Thresholds
CREATE TABLE platform_thresholds (
  id TEXT PRIMARY KEY,
  platform_id TEXT REFERENCES platforms(id),
  metric TEXT NOT NULL,            -- "frequency_7d", "ctr", "cpm"
  warning_value DECIMAL,           -- 3.0
  critical_value DECIMAL,          -- 5.0
  direction TEXT NOT NULL,         -- "above" or "below"
  context TEXT,                    -- "For retargeting campaigns"
  tactic_id TEXT,                  -- Optional: specific to a tactic
  created_at TIMESTAMP DEFAULT NOW()
);

-- Buyer Notes (crowdsourced knowledge)
CREATE TABLE platform_buyer_notes (
  id TEXT PRIMARY KEY,
  platform_id TEXT REFERENCES platforms(id),
  note_type TEXT NOT NULL,         -- "tip", "warning", "optimization", "gotcha"
  content TEXT NOT NULL,
  tactic_id TEXT,                  -- Optional: specific to a tactic
  contributed_by TEXT NOT NULL,
  upvotes INTEGER DEFAULT 0,
  is_verified BOOLEAN DEFAULT false,
  verified_by TEXT,
  created_at TIMESTAMP DEFAULT NOW()
);
```

## 2. Industry Knowledge Manager

This is your **research hub**â€”manually curated but structured for AI consumption.

### Database Schema

```sql
-- Industries
CREATE TABLE industries (
  id TEXT PRIMARY KEY,
  name TEXT NOT NULL,              -- "Automotive"
  code TEXT UNIQUE NOT NULL,       -- "automotive"
  description TEXT,
  icon TEXT,
  is_active BOOLEAN DEFAULT true,
  created_at TIMESTAMP DEFAULT NOW(),
  updated_at TIMESTAMP DEFAULT NOW()
);

-- Industry Benchmarks (the numbers)
CREATE TABLE industry_benchmarks (
  id TEXT PRIMARY KEY,
  industry_id TEXT REFERENCES industries(id),
  platform_id TEXT REFERENCES platforms(id),
  tactic_id TEXT,                  -- Optional: specific tactic
  metric TEXT NOT NULL,            -- "cpm", "ctr", "cpa"
  p25 DECIMAL,                     -- 25th percentile
  p50 DECIMAL NOT NULL,            -- Median (required)
  p75 DECIMAL,                     -- 75th percentile
  sample_size INTEGER,
  confidence DECIMAL,              -- 0-1
  quarter TEXT NOT NULL,           -- "Q1-2024"
  source TEXT,                     -- Where this data came from
  notes TEXT,
  created_at TIMESTAMP DEFAULT NOW(),
  updated_at TIMESTAMP DEFAULT NOW()
);

-- Industry Insights (the context)
CREATE TABLE industry_insights (
  id TEXT PRIMARY KEY,
  industry_id TEXT REFERENCES industries(id),
  insight_type TEXT NOT NULL,      -- "seasonality", "behavior", "competitive", "regulatory"
  title TEXT NOT NULL,
  content TEXT NOT NULL,
  ai_instruction TEXT,             -- How AI should apply this
  source TEXT,
  source_url TEXT,
  valid_from DATE,
  valid_until DATE,                -- For time-sensitive insights
  created_at TIMESTAMP DEFAULT NOW(),
  updated_at TIMESTAMP DEFAULT NOW()
);

-- Industry Seasonality (structured)
CREATE TABLE industry_seasonality (
  id TEXT PRIMARY KEY,
  industry_id TEXT REFERENCES industries(id),
  period_type TEXT NOT NULL,       -- "month", "quarter", "event"
  period_value TEXT NOT NULL,      -- "Q4", "December", "Black Friday"
  impact TEXT NOT NULL,            -- "peak", "low", "moderate"
  cpm_modifier DECIMAL,            -- 1.4 = 40% higher CPMs
  description TEXT,
  created_at TIMESTAMP DEFAULT NOW()
);
```

## 3. Soul Document Manager (System Prompts + Skills)

This is where it gets funâ€”**version-controlled, diffable, testable** prompt management.

### Database Schema

```sql
-- Soul Documents (system prompts, personas, guidelines)
CREATE TABLE soul_documents (
  id TEXT PRIMARY KEY,
  doc_type TEXT NOT NULL,          -- "system_prompt", "agent_persona", "output_guideline", "skill"
  name TEXT NOT NULL,              -- "Core Analyst Persona"
  slug TEXT UNIQUE NOT NULL,       -- "core-analyst-persona"
  description TEXT,
  is_active BOOLEAN DEFAULT true,
  created_at TIMESTAMP DEFAULT NOW(),
  updated_at TIMESTAMP DEFAULT NOW()
);

-- Soul Document Versions (the magic - full version history)
CREATE TABLE soul_document_versions (
  id TEXT PRIMARY KEY,
  document_id TEXT REFERENCES soul_documents(id),
  version INTEGER NOT NULL,
  content TEXT NOT NULL,           -- The actual prompt/skill content
  change_summary TEXT,             -- "Added healthcare compliance section"
  changed_by TEXT NOT NULL,
  is_published BOOLEAN DEFAULT false,
  published_at TIMESTAMP,
  created_at TIMESTAMP DEFAULT NOW(),
  
  UNIQUE(document_id, version)
);

-- Soul Document Tests (save test cases)
CREATE TABLE soul_document_tests (
  id TEXT PRIMARY KEY,
  document_id TEXT REFERENCES soul_documents(id),
  name TEXT NOT NULL,              -- "Automotive retargeting analysis"
  input_data JSONB NOT NULL,       -- Test input
  expected_behavior TEXT,          -- What should happen
  created_at TIMESTAMP DEFAULT NOW()
);

-- Soul Document Test Results (track how changes affect output)
CREATE TABLE soul_document_test_results (
  id TEXT PRIMARY KEY,
  test_id TEXT REFERENCES soul_document_tests(id),
  version_id TEXT REFERENCES soul_document_versions(id),
  output TEXT NOT NULL,            -- AI output for this version
  tokens_used INTEGER,
  latency_ms INTEGER,
  rating INTEGER,                  -- Manual rating 1-5
  notes TEXT,
  created_at TIMESTAMP DEFAULT NOW()
);
```
